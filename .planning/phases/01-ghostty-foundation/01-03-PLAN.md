use std::sync::atomic::{AtomicBool, Ordering};

/// Coalesces burst wakeup calls into a single GLib idle dispatch.
/// GLib's idle_add does not deduplicate — this flag prevents queueing N
/// idle tasks when Ghostty fires N wakeups in a single frame burst.
pub(crate) static WAKEUP_PENDING: AtomicBool = AtomicBool::new(false);

/// The GhosttyApp handle — stored as usize to be Send across the idle closure.
/// Safety: ghostty_app_t is opaque void* and only called from the GLib main thread.
pub(crate) static APP_PTR: std::sync::atomic::AtomicUsize =
    std::sync::atomic::AtomicUsize::new(0);

/// Called by Ghostty from its renderer thread. Must not call any ghostty_* API inline.
/// Instead, schedules ghostty_app_tick() on the GLib main loop (per D-04, GHOST-07).
pub extern "C" fn wakeup_cb(_userdata: *mut std::ffi::c_void) {
    // Swap: if already pending, another idle task is queued — skip.
    if WAKEUP_PENDING.swap(true, Ordering::SeqCst) {
        return;
    }
    glib::idle_add_once(|| {
        WAKEUP_PENDING.store(false, Ordering::SeqCst);
        let app_ptr = APP_PTR.load(Ordering::SeqCst);
        if app_ptr != 0 {
            unsafe {
                let app = app_ptr as crate::ghostty::ffi::ghostty_app_t;
                crate::ghostty::ffi::ghostty_app_tick(app);
            }
        }
    });
}

/// Called by Ghostty when a surface wants to close (e.g. shell exits).
/// Runs on the GLib main thread (called during ghostty_app_tick).
pub extern "C" fn close_surface_cb(
    _userdata: *mut std::ffi::c_void,
    surface: crate::ghostty::ffi::ghostty_surface_t,
) {
    // Phase 1: single surface. When the surface closes, free it and quit the app.
    unsafe {
        crate::ghostty::ffi::ghostty_surface_free(surface);
    }
    // Quit the GTK application if the only terminal surface closes.
    if let Some(app) = gtk4::prelude::ApplicationExt::default() {
        gtk4::prelude::GtkApplicationExt::quit(&app);
    }
}

    // WAKEUP_PENDING must be false at program start (or after reset)
    cmux_linux::ghostty::callbacks::WAKEUP_PENDING.store(false, std::sync::atomic::Ordering::SeqCst);
    assert!(!cmux_linux::ghostty::callbacks::WAKEUP_PENDING.load(std::sync::atomic::Ordering::SeqCst));

#[test]
fn wakeup_coalescing_flag_set_on_first_call() {
    // After first wakeup_cb fires and sets the flag, a second swap returns true (already pending)
    use std::sync::atomic::Ordering;
    cmux_linux::ghostty::callbacks::WAKEUP_PENDING.store(false, Ordering::SeqCst);

    // Simulate first wakeup: swap(true) returns false → schedule idle

    // Reset for other tests
    cmux_linux::ghostty::callbacks::WAKEUP_PENDING.store(false, Ordering::SeqCst);

**Full implementation:**

```rust
use std::cell::Cell;
use std::ffi::CString;
use std::sync::atomic::Ordering;
use gtk4::prelude::*;
use glib;

use crate::ghostty::ffi;
use crate::ghostty::callbacks::{wakeup_cb, close_surface_cb, action_cb, APP_PTR, WAKEUP_PENDING};

/// GhosttyWidget initializes a GtkGLArea + ghostty surface pair and wires all signals.
/// Returns the GtkGLArea widget to insert into the window.
///
/// Safety: All ghostty_* calls happen on the GLib main thread (in signal handlers or
/// glib::idle callbacks). ghostty_app_t and ghostty_surface_t are opaque pointers
/// never shared across threads — conforming to D-10 (Phase 3 socket compatibility).
pub fn create_surface(app: &gtk4::Application) -> gtk4::GLArea {
    let gl_area = gtk4::GLArea::new();
    // Per Pitfall 1: require OpenGL 4.3 before the area is realized.
    gl_area.set_required_version(4, 3);
    // Manual render mode: only render when wakeup_cb schedules queue_render().
    // An independent render loop adds input latency (per CLAUDE.md pitfall).
    gl_area.set_auto_render(false);

    // Initialize ghostty app (one-time, before surface creation).
    // Safety: called once on the main thread before any surface is created.
    let ghostty_app = unsafe {
        let argv: Vec<CString> = std::env::args()
            .map(|a| CString::new(a).unwrap())
            .collect();
        let mut ptrs: Vec<*mut i8> = argv.iter().map(|a| a.as_ptr() as *mut i8).collect();
        ffi::ghostty_init(ptrs.len() as i32, ptrs.as_mut_ptr());

        let config = ffi::ghostty_config_new();
        ffi::ghostty_config_load_default_files(config);
        ffi::ghostty_config_finalize(config);  // CRITICAL: finalize before ghostty_app_new

        // Runtime config: register all C callbacks.
        // userdata is set to the app pointer after creation (chicken-and-egg).
        let runtime_config = ffi::ghostty_runtime_config_s {
            userdata: std::ptr::null_mut(),  // updated below after app creation
            supports_selection_clipboard: 1,
            wakeup_cb: Some(wakeup_cb),
            action_cb: Some(action_cb),
            read_clipboard_cb: Some(read_clipboard_cb),
            confirm_read_clipboard_cb: Some(confirm_read_clipboard_cb),
            write_clipboard_cb: Some(write_clipboard_cb),
            close_surface_cb: Some(close_surface_cb),
        };

        let ghostty_app = ffi::ghostty_app_new(&runtime_config as *const _ as *mut _, config);
        ffi::ghostty_config_free(config);

        // Store app pointer globally for wakeup_cb to use.
        APP_PTR.store(ghostty_app as usize, Ordering::SeqCst);

        ghostty_app
    };

    // Clone gl_area reference for use in closures.
    let gl_area_clone = gl_area.clone();
    let gl_area_for_realize = gl_area.clone();

    // Create the ghostty surface config pointing at this GtkGLArea.
    // Surface creation happens before realize — the PTY starts immediately.
    // Rendering is deferred until realize (Pitfall 4 guard in wakeup path).
    let ghostty_surface = unsafe {
        let gl_area_ptr = gl_area.as_ptr() as *mut std::ffi::c_void;

        let platform = ffi::ghostty_platform_u {
            gtk4: ffi::ghostty_platform_gtk4_s { gl_area: gl_area_ptr },
        };

        let mut surface_config = std::mem::zeroed::<ffi::ghostty_surface_config_s>();
        surface_config.platform_tag = ffi::GHOSTTY_PLATFORM_GTK4 as i32;
        surface_config.platform = platform;
        surface_config.userdata = std::ptr::null_mut();
        surface_config.scale_factor = 1.0;  // updated in realize signal

        ffi::ghostty_surface_new(ghostty_app, &surface_config)
    };

    // ── GtkGLArea::realize ───────────────────────────────────────────────────
    // GL context is now valid. Set size, scale, and focus.
    gl_area.connect_realize({
        let surface = ghostty_surface;
        move |area| {
            area.make_current();
            if let Some(err) = area.error() {
                eprintln!("cmux: GLArea realize error: {err}");
                std::process::exit(1);  // Per D-09: no GUI error dialog in Phase 1
            }
            let scale = area.scale_factor() as f64;
            let w = area.width();
            let h = area.height();
            unsafe {
                // Per Pitfall 5: convert logical→physical pixels.
                ffi::ghostty_surface_set_size(
                    surface,
                    (w as f64 * scale) as u32,
                    (h as f64 * scale) as u32,
                );
                ffi::ghostty_surface_set_content_scale(surface, scale, scale);
                ffi::ghostty_surface_set_focus(surface, true);
            }
        }
    });

    // ── GtkGLArea::render ────────────────────────────────────────────────────
    // Called by GTK frame clock when queue_render() was requested.
    gl_area.connect_render({
        let surface = ghostty_surface;
        move |_area, _ctx| {
            unsafe { ffi::ghostty_surface_draw(surface); }
            glib::Propagation::Stop  // suppress GTK default render
        }
    });

    // ── GtkGLArea::resize ────────────────────────────────────────────────────
    // GTK provides logical (CSS) pixels. Ghostty needs physical pixels (Pitfall 5).
    gl_area.connect_resize({
        let surface = ghostty_surface;
        move |area, logical_w, logical_h| {
            let scale = area.scale_factor();
            unsafe {
                ffi::ghostty_surface_set_size(
                    surface,
                    (logical_w * scale) as u32,
                    (logical_h * scale) as u32,
                );
            }
        }
    });

    // ── notify::scale-factor (GHOST-06) ─────────────────────────────────────
    // Fires when the window moves to a monitor with a different DPI.
    gl_area.connect_notify(Some("scale-factor"), {
        let surface = ghostty_surface;
        move |widget, _| {
            let scale = widget.scale_factor() as f64;
            unsafe {
                ffi::ghostty_surface_set_content_scale(surface, scale, scale);
                ffi::ghostty_surface_refresh(surface);  // trigger redraw at new scale
            }
        }
    });

    // ── Wakeup → queue_render (GHOST-07) ────────────────────────────────────
    // wakeup_cb calls ghostty_app_tick() via glib::idle_add_once.
    // After ghostty_app_tick(), we need to queue_render() on the GtkGLArea.
    // Wire this by connecting a second idle from the wakeup path.
    // Approach: override wakeup path in APP_PTR; the tick + render are coupled
    // by having ghostty_app_tick trigger the render callback internally via
    // the GtkGLArea render signal. Alternatively, the wakeup idle callback
    // calls queue_render() after ghostty_app_tick().
    //
    // Simplest correct approach: modify wakeup_cb (callbacks.rs) to also
    // call gl_area.queue_render() after the tick IF gl_area is realized.
    // Since callbacks.rs doesn't know about the gl_area, we use a thread_local
    // or a global Weak<GLArea> ref. For Phase 1 (single surface), use a global.

    // Store gl_area weak pointer for wakeup_cb to call queue_render on.
    // This is safe: accessed only from GLib main thread (idle_add_once).
    GL_AREA_FOR_RENDER.with(|cell| {
        cell.set(Some(gl_area_clone));
    });

    gl_area
}

// Thread-local storage for the GLArea — allows wakeup_cb (via idle_add_once)
// to call queue_render() on the main thread without passing it through C callbacks.
thread_local! {
    pub(crate) static GL_AREA_FOR_RENDER: Cell<Option<gtk4::GLArea>> = Cell::new(None);
}
```

extern "C" fn read_clipboard_cb(
    _userdata: *mut std::ffi::c_void,
    clipboard_type: i32,
    surface: ffi::ghostty_surface_t,

        Some(d) => d,
        None => return,

    let clipboard = if clipboard_type == ffi::GHOSTTY_CLIPBOARD_SELECTION as i32 {

extern "C" fn write_clipboard_cb(
    _userdata: *mut std::ffi::c_void,
    clipboard_type: i32,

    _len: usize,
