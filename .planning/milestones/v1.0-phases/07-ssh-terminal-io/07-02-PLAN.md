**1. Add RemoteOutput/RemoteEof/StreamOpened handling to main.rs SSH event processing:**

Find the existing SSH event processing code (likely in a `glib::MainContext::default().spawn_local()` or a polling timer that reads from `ssh_rx`). Add handling for the new variants:

```rust
SshEvent::RemoteOutput { pane_id, data } => {
    // Look up the Ghostty surface for this pane_id.
    // Use SURFACE_REGISTRY to find surface pointer by pane_id.
    if let Ok(registry) = crate::ghostty::callbacks::SURFACE_REGISTRY.lock() {
        // SURFACE_REGISTRY maps surface_ptr -> pane_id. Need reverse lookup.
        let surface_ptr = registry.iter()
            .find(|(_, &pid)| pid == pane_id)
            .map(|(&sptr, _)| sptr as ffi::ghostty_surface_t);
        if let Some(surface) = surface_ptr {
            unsafe {
                ffi::ghostty_surface_process_output(
                    surface,
                    data.as_ptr() as *const _,
                    data.len(),
                );
            }
            // Queue render for the GLArea
            if let Ok(gl_areas) = crate::ghostty::callbacks::GL_TO_SURFACE.lock() {
                for (&gl_ptr, &s_ptr) in gl_areas.iter() {
                    if s_ptr == surface as usize {
                        let area: glib::translate::Borrowed<gtk4::GLArea> =
                            unsafe { glib::translate::from_glib_borrow(gl_ptr as *mut gtk4::ffi::GtkGLArea) };
                        area.queue_render();
                        break;
                    }
                }
            }
        }
    }
}

SshEvent::RemoteEof { pane_id } => {
    // Per D-08: write exit message to surface, keep pane open
    if let Ok(registry) = crate::ghostty::callbacks::SURFACE_REGISTRY.lock() {
        let surface_ptr = registry.iter()
            .find(|(_, &pid)| pid == pane_id)
            .map(|(&sptr, _)| sptr as ffi::ghostty_surface_t);
        if let Some(surface) = surface_ptr {
            let msg = b"\r\n\x1b[90m[Remote shell exited. Press any key to close]\x1b[0m\r\n";
            unsafe {
                ffi::ghostty_surface_process_output(surface, msg.as_ptr() as *const _, msg.len());
            }
        }
    }
}

SshEvent::StreamOpened { pane_id, stream_id } => {
    // Set the stream_id on the IoWriteContext so keystrokes start flowing
    if let Some(ctx) = app_state.borrow().remote_pane_contexts.get(&pane_id) {
        if let Ok(mut sid) = ctx.stream_id.lock() {
            *sid = Some(stream_id);
        }
    }
}
```
